# Task: Replace shadcn Badge and Button Components

Replace all instances of shadcn/ui Badge and Button components with custom components from Task 002. These are used throughout the app for method badges (GET, POST, etc.) and action buttons (copy, delete, etc.).

## Technical Details

### Files to Modify

**1. RequestList component** - Method badges
```tsx
// Before
import { Badge } from '@/components/ui/badge'

<Badge variant={getMethodColor(request.method)}>
  {request.method}
</Badge>

// After
import { Badge } from '@/components/ui'

<Badge variant={getMethodVariant(request.method)}>
  {request.method}
</Badge>

// Update helper function
function getMethodVariant(method: string): 'success' | 'info' | 'warning' | 'error' {
  switch (method.toUpperCase()) {
    case 'GET': return 'success'
    case 'POST': return 'info'
    case 'PUT':
    case 'PATCH': return 'warning'
    case 'DELETE': return 'error'
    default: return 'default'
  }
}
```

**2. RequestDetail component** - Copy buttons
```tsx
// Before
import { Button } from '@/components/ui/button'

<Button variant="outline" size="sm" onClick={handleCopy}>
  <Copy className="h-4 w-4" />
  Copy
</Button>

// After
import { Button } from '@/components/ui'

<Button variant="secondary" size="sm" onClick={handleCopy}>
  <Copy className="h-4 w-4" />
  Copy
</Button>
```

**3. EndpointView component** - Action buttons
```tsx
// Before
<Button variant="destructive" onClick={handleDelete}>
  Delete Endpoint
</Button>

// After
<Button variant="danger" onClick={handleDelete}>
  Delete Endpoint
</Button>
```
