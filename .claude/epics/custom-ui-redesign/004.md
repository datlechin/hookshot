Replace all instances of shadcn/ui Dialog component with the custom Modal component from Task 003. Primary focus is the RequestDetail component which uses Dialog to show webhook request details.

## Technical Details

### Files to Modify

**`frontend/src/components/RequestDetail.tsx`**

Before (shadcn Dialog):
```tsx
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
} from '@/components/ui/dialog'

export function RequestDetail({ request, onClose }: RequestDetailProps) {
  return (
    <Dialog open={!!request} onOpenChange={onClose}>
      <DialogContent className="max-w-4xl max-h-[85vh]">
        <DialogHeader>
          <DialogTitle>...</DialogTitle>
          <DialogDescription>...</DialogDescription>
        </DialogHeader>
        {/* Content */}
      </DialogContent>
    </Dialog>
  )
}
```

After (custom Modal):
```tsx
import { Modal } from '@/components/ui'

export function RequestDetail({ request, onClose }: RequestDetailProps) {
  if (!request) return null

  return (
    <Modal
      isOpen={!!request}
      onClose={onClose}
      title={`Request #${request.id}`}
      size="xl"
    >
      <div className="text-xs text-secondary mb-4">
        {new Date(request.received_at).toLocaleString()} • IP: {request.ip_address}
      </div>
      {/* Rest of content */}
    </Modal>
  )
}
```
